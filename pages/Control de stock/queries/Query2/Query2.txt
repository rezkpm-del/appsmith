SELECT 
    IdArticulo,
    SUM(CantidadVendida) as cant_vendida
FROM stock_resumenes
WHERE IdSucursal = {{sel_sucursal.selectedOptionValue}}
-- Usamos la fecha contable directamente de esta tabla
AND FechaContable >= {{moment(fecha_inicio.selectedDate).format('YYYY-MM-DD')}}
AND FechaContable <= {{moment(fecha_fin.selectedDate).format('YYYY-MM-DD')}}
-- Mantenemos el filtro de insumo que ya logramos hacer andar
AND ({{sel_insumo.selectedOptionValue || -1}} = -1 OR IdArticulo = {{sel_insumo.selectedOptionValue || -1}})
GROUP BY IdArticulo;
